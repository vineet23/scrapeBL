  <script>
   /**
     Usage:
     set variable "defaultOption" for each wp-install:
     0: show per 100g/ml
     1: show per portion
     2: show per (fl.) oz.
     for fluids: add 'data-fluids="fluids"' to the table tag
     for OZ: add '<td class="serving oz">Serving OZ</td>' to the table head and
     '<td class="serving oz">1 oz. (<data>28,5</data> g)</td>' to every table row; !BOTH AFTER the td with class 'serving portion'
     */

    var disableKj = true;

    var defaultOption = '1'; //Set only this one
    var isfluid = false;
    var hasOz = false;
    var tblSelector = '#calories-table';
    var tblObj = null;
    var tbl = null;

    function getDataField(x) {
        return parseInt(jQuery(x).html());
    }

    jQuery(document).ready(function () {
        tbl = jQuery(tblSelector);
        if (tbl.length <= 0) {
            return;
        }
        hasOz = (tbl.find('tbody tr td.oz').length > 0);
        isfluid = (tbl.attr('data-fluids') == 'fluids');
        //Prepare Table and extra columns
        tbl.find('thead tr').append('<td class="kcal kcal-serving">Calories</td><td class="kj kj-serving">Kilojoule</td>' + (hasOz ? '<td class="kcal kcal-oz">Calories</td><td class="kj kj-oz">Kilojoule</td>' : ''));
        tbl.find('tbody tr').each(function () {
            var me = jQuery(this);
            var portion = getDataField(me.find('td.portion').html());
            var kcal = getDataField(me.find('td.kcal').html());
            var kj = getDataField(me.find('td.kj').html());
            me.append('<td class="kcal portion"><data>' + Math.round(portion / 100 * kcal) + '</data> cal</td>');
            me.append('<td class="kj portion"><data>' + Math.round(portion / 100 * kj) + '</data> kJ</td>');
            if (hasOz) {
                var oz = getDataField(me.find('td.oz').html());
                me.append('<td class="kcal oz"><data>' + Math.round(oz / 100 * kcal) + '</data> cal</td>');
                me.append('<td class="kj oz"><data>' + Math.round(oz / 100 * kj) + '</data> kJ</td>');
            }
        });
        //Adapt CSS
        jQuery('table#calories-table td.portion').css('display', 'table-cell');
        //Get Columns Info
        var aoColumns = [
            null,
            {"type": "parseData", "orderable": false},
            {"type": "parseData"},
            {"type": "parseData"},
            {"type": "parseData"},
            {"type": "parseData"},
            {"type": "parseData"}
        ];
        if (hasOz) {
            aoColumns.push({"type": "parseData"});
            aoColumns.push({"type": "parseData"});
            aoColumns.push({"type": "parseData"});
        }
        //Set the data table
        tblObj = tbl.dataTable({
            "paging": false,
            "info": false,
            "lengthChange": false,
            "columns": aoColumns,
            //"aaSorting": [], //add for no element order on load
            "autoWidth": false,
            "searching": true,
            "language": {  // Disable the search label and add placeholder
                search: "_INPUT_",
                searchPlaceholder: "Search category",
                zeroRecords: "No results found. <a id=calories-search-site href=#>Search the entire website</a>"
            }
        });
        showServing(defaultOption);

        var disableOunces = '0';

        // Setup translations
        var txtVal1 = "show per 100";
        var txtVal2 = "show per serving";
        var txtVal3 = "show per";
        var txtMl = "ml";
        var txtG = "g";
        var txtFluid = "fluid";
        var txtOunces = "ounces";


        // Setup the filter markup
        var markup = '<div class=calorie-filter>';
        markup += '<div class=calorie-filter-label>Please choose your serving size</div>';
        markup += '<div class=calorie-filter-buttons>';
        markup += '<div data class="calorie-filter-button ' + (defaultOption == 0 ? 'filter_selected' : '') + '">' + txtVal1 + ' ' + (isfluid ? txtMl : txtG) + '</div>';
        markup += '<div data=1 class="calorie-filter-button ' + (defaultOption == 1 ? 'filter_selected' : '') + '">' + txtVal2 + '</div>';
        if (hasOz && disableOunces == 0) {
            markup += '<div data=2 class="calorie-filter-button ' + (defaultOption == 2 ? 'filter_selected' : '') + '">' + txtVal3 + ' ' + (isfluid ? txtFluid : '') + ' ' + txtOunces + '</div>';
        }
        markup += '</div>';

        // Append markup
        jQuery('#table-filter').append(jQuery(markup));

        // Handle clicking on the filters
        jQuery(document).on('click', '.calorie-filter-button', function(){
            $ = jQuery;
            var val = $(this).attr('data');
            var $parent = $(this).closest('.calorie-filter');
            $parent.find('.filter_selected').removeClass('filter_selected');
            $(this).addClass('filter_selected');
            showServing(val);

            // Analytics
            ga("send", "event", {
                "eventCategory": 'CALORIETABLES: Pizza',
                "eventAction": "Calories table dropdown change",
                "eventLabel": $(this).text(),
                "nonInteraction": false
            });
        });

        // Handle empty search filter
        var dt = jQuery(tblSelector).DataTable();
        dt.on('search.dt', function(e, settings){
            var filter = jQuery(e.target).closest('.dataTables_wrapper').find('input[type=search]');
            var data = dt.rows( { filter : 'applied'} ).nodes();
            withDelay(function(){
                if(data.length === 0) {
                    // Analytics
                    ga("send", "event", {
                        "eventCategory": 'CALORIETABLES: Pizza',
                        "eventAction": filter.val(),
                        "eventLabel": 'Datatable Empty',
                        "nonInteraction": false
                    });
                }
            }, 1000)
        });

        // If kj column is disabled, hide it.
        if (disableKj) {
            var $kj = jQuery(document).find('td.kj');
            $kj.hide();
            $kj.closest('table').addClass('calorie-hide-kj')
        }

        // Add X button to the datatable
        jQuery('.dataTables_filter label').append('<a id=calories-clear-saarch style=display:none; href=#><i class="fa fa-times"></i></a>')
        jQuery(document).on('input', '.dataTables_filter label input', function(){
            var searchControl = jQuery('#calories-clear-saarch');
            var value = jQuery(this).val();
            if(value && value != '') {
                searchControl.show();
            } else {
                searchControl.hide();
            }
        });
        jQuery(document).on('click', '#calories-clear-saarch', function(e){
            e.preventDefault();
            jQuery('.dataTables_filter label input').val('').focus();
            jQuery(this).hide();
            tblObj.DataTable().search('').draw();
        });
    });



    //Some Table Sort Functions
    jQuery.fn.dataTableExt.oSort['parseData-asc'] = function (x, y) {
        x = getDataField(x);
        y = getDataField(y);
        return ((x < y) ? -1 : ((x > y) ? 1 : 0));
    };
    jQuery.fn.dataTableExt.oSort['parseData-desc'] = function (x, y) {
        x = getDataField(x);
        y = getDataField(y);
        return ((x < y) ? 1 : ((x > y) ? -1 : 0));
    };
    var firstEffect = false;

    //Show Hide Servingsinput-placeholder
    function showServing(what) {
        var offset = hasOz ? 1 : 0;
        //100g
        tblObj.fnSetColumnVis(1, what == 0);
        tblObj.fnSetColumnVis(3 + offset, what == 0);
        tblObj.fnSetColumnVis(4 + offset, what == 0);
        //Serving
        tblObj.fnSetColumnVis(2, what == 1);
        tblObj.fnSetColumnVis(5 + offset, what == 1);
        tblObj.fnSetColumnVis(6 + offset, what == 1);
        //OZ
        if (hasOz) {
            tblObj.fnSetColumnVis(3, what == 2);
            tblObj.fnSetColumnVis(7 + offset, what == 2);
            tblObj.fnSetColumnVis(8 + offset, what == 2);
        }
        //jQuery(tblSelector).find('thead td').attr('style','');
        //Effect
        if (firstEffect) {
            var effectOn = tbl.find('tbody .serving, tbody .kcal, tbody .kj');
            effectOn.stop(true, true);
            effectOn.animate({
                "color": '#FFD700' //Highlight
            }, 500, function () {
                effectOn.animate({
                    "color": '#565656' //Back
                }, 500, function () {
                });
            });
        }
        firstEffect = true;
    }

How the pizza crust is made will also affect the number of calories and overall health benefits. Using whole wheat flour instead of white flour increases the amount of complex carbs. This also increases nutrient density, and whole grains are linked to reduced risk of stroke and type 2 diabetes. What you put on top of the pizza makes the biggest difference in overall nutrition of the slice youâ€™re eating. High-calorie toppings include cheese, greasy sausage, alfredo (white) sauce, and salamis such as the ever-popular pepperoni. Pizzas with less calories are topped with lots of veggies like peppers, fresh greens like spinach, and mushrooms. Because the number of calories in pizza can vary so widely, this calorie chart should be used as a general guideline, except in the case of specific restaurant items, for example Pizza Hut.
